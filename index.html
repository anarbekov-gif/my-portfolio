from tkinter import messagebox
from deep_translator import GoogleTranslator

def translate_text(event=None):
    text = input_text.get("1.0", "end-1c").strip()
    if not text:
        return "break"

    try:
        target = 'en' if direction_var.get() == 'RU ‚Üí EN' else 'ru'
        translated = GoogleTranslator(source='auto', target=target).translate(text)

        output_text.config(state=tk.NORMAL)
        output_text.delete("1.0", tk.END)
        output_text.insert("1.0", translated)
        output_text.config(state=tk.DISABLED)
        return "break"
    except Exception:
        messagebox.showerror("–û—à–∏–±–∫–∞", "–ü—Ä–æ–≤–µ—Ä—å—Ç–µ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ –∏–Ω—Ç–µ—Ä–Ω–µ—Ç—É")
        return "break"

def swap_direction():

    new = 'EN ‚Üí RU' if current == 'RU ‚Üí EN' else 'RU ‚Üí EN'
    direction_var.set(new)
